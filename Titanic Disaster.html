def map_Embarked(e):
    if e == 'C':
        return 0
    elif e == 'Q':
        return 1
    elif e == 'S':
        return 2

def map_Sex(sex):
    if sex == 'male':
        return 0
    elif sex == 'female':
        return 1
    
train_df['Embarked'] = train_df.Embarked.apply(lambda e : map_Embarked(e))
train_df['Sex'] = train_df.Sex.apply(lambda sex : map_Sex(sex))
